7.1 Learning Outcome

pacman::p_load(tidyverse, ggExtra, ggiraph)
exam_data <- read_csv("data/Exam_data.csv")

A simple scatterplot

p <- ggplot(data=exam_data, 
            aes(x= MATHS, y=ENGLISH)) +
  geom_point() +
  coord_cartesian(xlim=c(0,100),
                  ylim=c(0,100))
p
